## Installation

```bash
npx shadcn@latest add @joyco/file-dropzone
```

## Usage

```tsx
import { FileDropzone } from '@/components/file-dropzone'
;<FileDropzone
  accept="image/png,image/jpeg"
  maxSizeMB={5}
  onUpload={handleUpload}
/>
```

## Usage

```tsx
import { ScrollAreaViewport, ScrollAreaContent } from '@/components/scroll-area'

<ScrollAreaViewport>
  <ScrollAreaContent>
    <div>Content here</div>
  </ScrollAreaContent>
</ScrollAreaViewport>
```

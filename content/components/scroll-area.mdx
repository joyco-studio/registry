## Usage

```tsx
import { ScrollArea } from '@/registry/joyco/blocks/scroll-area'

function App() {
  return (
    <ScrollArea.Root className="h-64">
      <ScrollArea.Content className="space-y-4 p-6">
        <div>Content here</div>
      </ScrollArea.Content>
    </ScrollArea.Root>
  )
}
```

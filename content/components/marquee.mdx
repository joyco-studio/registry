## Installation

```bash
npx shadcn@latest add https://registry.joyco.studio/r/marquee.json
```

## Usage

```tsx
import { Marquee } from '@joyco/marquee'

function App() {
  return (
    <Marquee speed={100} speedFactor={1} direction={1}>
      <div>Your content here</div>
    </Marquee>
  )
}
```

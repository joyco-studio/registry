## Installation

```bash
npx shadcn@latest add https://r.joyco.studio/chat.json
```

## Usage

```tsx
import {
  Chat,
  ChatInputArea,
  ChatInputField,
  ChatInputSubmit,
  ChatViewport,
  ChatMessages,
  ChatMessageRow,
  ChatMessageAvatar,
  ChatMessageBubble,
  ChatMessageTime,
} from '@/components/chat'

<Chat>
  <ChatViewport>
    <ChatMessages>
      <ChatMessageRow>
        <ChatMessageAvatar />
        <ChatMessageBubble />
        <ChatMessageTime />
      </ChatMessageRow>
    </ChatMessages>
  </ChatViewport>

  <ChatInputArea>
    <ChatInputField />
    <ChatInputSubmit />
  </ChatInputArea>
</Chat>
```

## Installation

```bash
npx shadcn@latest add https://registry.joyco.studio/r/mobile-menu.json
```

## Usage

```tsx
import {
  MobileMenu,
  MobileMenuTrigger,
  MobileMenuContent,
  MobileMenuHeader,
  MobileMenuNav,
  MobileMenuLink,
  MobileMenuClose,
} from '@/components/ui/mobile-menu'
import { Menu, X } from 'lucide-react'

function App() {
  return (
    <MobileMenu>
      <MobileMenuTrigger>
        <Menu className="h-6 w-6" />
      </MobileMenuTrigger>
      <MobileMenuContent>
        <MobileMenuHeader>
          <span>Menu</span>
          <MobileMenuClose>
            <X className="h-6 w-6" />
          </MobileMenuClose>
        </MobileMenuHeader>
        <MobileMenuNav>
          <MobileMenuLink href="/">Home</MobileMenuLink>
          <MobileMenuLink href="/about">About</MobileMenuLink>
        </MobileMenuNav>
      </MobileMenuContent>
    </MobileMenu>
  )
}
```

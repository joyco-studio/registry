## Installation

```bash
npx shadcn@latest add @joyco/infinite-list
```

## Usage

```tsx
import { useInfiniteList, MaskedList } from '@/components/infinite-list'
;<MaskedList {...list}>
  {items.map((item) => (
    <Item key={item.id} {...item} />
  ))}
</MaskedList>
```

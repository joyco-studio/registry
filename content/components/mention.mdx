## Installation

```bash
npx shadcn@latest add @joyco/mention
```

## Usage

```tsx
import {
  Mention,
  MentionInput,
  MentionList,
  MentionItem,
  MentionItemText,
} from '@/components/ui/mention'
;<Mention>
  <MentionInput />
  <MentionList>
    <MentionItem>
      <MentionItemText />
    </MentionItem>
  </MentionList>
</Mention>
```
